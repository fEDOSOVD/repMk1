#include <iostream>
#include <fstream>
#include <string>
#include <vector>

using namespace std;

int main() {
    // Шаблоны дней недели
    const vector<string> days = { "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday" };
    // Шаблоны предметов
    const vector <string> subjects = { "Math", "Physics", "Chemistry", "Biology", "History" };

    // Создание текстовых файлов для каждого дня недели
    for (int i = 0; i < 7; ++i) {
        // Открываем файл с именем дня недели (например, "Monday.txt")
        fstream file(days[i] + ".txt");
        // Записываем названия предметов в файл
        for (const auto& subject : subjects) {
            file << subject << endl; // Записываем каждый предмет на новой строке
        }
        file.close(); // Закрываем файл после записи
    }

    // Открываем созданные файлы в текстовом редакторе
    for (const auto& day : days) {
        // Формируем команду для открытия файла с помощью
        string command =  day + ".txt";
        // Выполняем команду, открывая файл
        system(command.c_str());
    }

    return 0; // Завершаем программу
}
